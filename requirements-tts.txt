# Kokoro-82M text-to-speech (Speech tab).
#
#   pip install -r requirements-tts.txt
#
# Installs hexgrad/Kokoro (Apache-2.0, ~82M params) with English G2P and WAV I/O.
# The model downloads from Hugging Face on first generation (~300 MB). Voices are
# bundled with the package.
#
# CPU works well (often faster than real-time on modern machines). A CUDA GPU is
# used automatically when available for extra speed.
#
# Optional: install espeak-ng for better out-of-vocabulary / non-English fallback
# pronunciation — English narration works without it via misaki[en].
#   Windows: https://github.com/espeak-ng/espeak-ng/releases (msi installer)
#   macOS:   brew install espeak-ng
#   Linux:   apt/yum install espeak-ng

kokoro>=0.9.4
soundfile>=0.12.1
misaki[en]>=0.9.4
